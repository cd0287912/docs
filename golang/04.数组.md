### GO 数组

数组是一系列同一类型的集合。
数组是一个长度固定的数据类型，数组的长度是类型的一部分。
数组的默认值是对应其数据类型的零值。

一维数组

```go
	// 定义数组  var 数组变量名字 [元素数量]T
	var ages [5]int
	fmt.Printf("ages type is %T\n", ages) // ages type is [5]int

	// 初始化数组方式1
	var names [5]string
	names[0] = "mike"
	names[1] = "jack"
	fmt.Println(names)

	// 初始化数组方式2
	var num [2]int = [2]int{1, 2}
	fmt.Println(num)

	// 初始化数组方式3 - 编译器推导数组长度
	var arr = [...]int{1, 2, 3, 4, 5}
	fmt.Println(arr)

	// 初始化数组方式4 - 指定其索引
	arr2 := [...]int{0: 10, 2: 20, 4: 40}
	fmt.Println(arr2) // [10 0 20 0 40]

	// 数组操作
	list := [...]string{"ka", "ko", "la"}
	fmt.Println("list length is ", len(list))
	list[2] = "ka"
	fmt.Println("list is", list)

	for i := 0; i < len(list); i++ {
		fmt.Print(list[i])
	}
	fmt.Println()
	for index, value := range list {
		fmt.Printf("index=%v,value=%v\n", index, value)
	}

	// go中 基本数据类型 和 数组都是【值传递】，切片是【引用传递】
	var arr1 = [...]int{1, 2, 3}
	arr2 := arr1
	arr1[0] = 5
	fmt.Println(arr1) // [5 2 3]
	fmt.Println(arr2) // [1 2 3]
```

二维数组

```go
	// 定义数组  var 数组变量名字 [元素数量][元素数量]T
	var arr = [3][2]string{
		{"mike", "jack"},
		{"lucy", "bob"},
		{"alan", "tom"},
	}
	fmt.Println(arr)
```
