### GO time

go中内置time包，来操作时间相关

时间对象操作
```go
func main() {
	// 获取当前时间
	now := time.Now()
	// 提取年月日
	year := now.Year()
	month := now.Month()
	day := now.Day()
	hour := now.Hour()
	min := now.Minute()
	second := now.Second()
	fmt.Println(year, month, day, hour, min, second)
	// 格式化输出
	str := now.Format("2006-01-02 15:04:05")
	fmt.Println(str)
	// 获取时间戳
	timestanp := now.Unix()
	fmt.Println(timestanp)
}
```

时间戳转时间对象
```go
func main() {
	// 当前时间戳 1758460180
	timeStamp := 1758460180
	// 时间戳转时间
	t := time.Unix(int64(timeStamp), 0)
	fmt.Println(t)
	str := t.Format("2006-01-02 15:04:05")
	fmt.Println(str)
}
```

时间文本转时间对象
```go
func main() {
	dateStr := "2025-09-21 21:09:40"
	// 时间文本转时间对象
	t, err := time.Parse("2006-01-02 15:04:05", dateStr)
	if err != nil {
		fmt.Println("时间格式错误")
		return
	}
	fmt.Println(t)
}
```

时间操作函数
```go
func main() {
	now := time.Now()
	nextHour := now.Add(time.Hour)
	fmt.Println(nextHour)
}
```
