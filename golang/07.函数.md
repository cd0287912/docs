### GO 函数

```go
// 函数的定义
// 固定参数
func subFn(x, y int) int {
	return x - y
}
// 可变参数- x是一个切片
func sumFn(x ...int) {
	var sum int
	for _, v := range x {
		sum += v
	}
	fmt.Println(sum)
}
// 一次返回多个值
func calc(x int, y int) (int, int) {
	sum := x + y
	sub := x - y
	return sum, sub
}
// 返回值命名，函数体内可以直接使用，不需要再声明
func calc(x int, y int) (sum int, sub int) {
	sum = x + y
	sub = x - y
	return
}
```

```go
// 练习：排序
func sortIntAsc(slice []int) []int {
	for i := 0; i < len(slice); i++ {
		for j := i + 1; j < len(slice); j++ {
			if slice[i] > slice[j] {
				temp := slice[j]
				slice[j] = slice[i]
				slice[i] = temp
			}
		}
	}
	return slice
}
```
